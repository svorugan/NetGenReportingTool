SELECT p.FULL_NAME, a.JOB_NAME, a.ORGANIZATION_NAME, a.PAYROLL_NAME
FROM PER_PEOPLE_AI_V p
JOIN PER_ASSIGNMENTS_AI_V a ON p.PERSON_ID = a.PERSON_ID
WHERE p.CURRENT_EMPLOYEE_FLAG = 'Y'
AND a.ASSIGNMENT_STATUS NOT IN ('Do Not Pay RU (DO NOT PROCESS)', 'LOA without Pay RU', 'LOAMA without Pay RU')
AND a.ASSIGNMENT_STATUS NOT LIKE '%Leave%'
AND a.ASSIGNMENT_STATUS NOT LIKE '%LOA%'
AND a.ASSIGNMENT_STATUS NOT LIKE '%without Pay%'
AND a.ASSIGNMENT_STATUS NOT LIKE '%w/o Pay%'
AND a.IS_PRIMARY_ASSIGNMENT = 'Y';
